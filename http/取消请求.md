```js
// define 定义
function query(options) {
    const controller = new AbortController();

    const { signal } = controller;

    const cancel = function () {
        // 副作用实现
        controller.abort();
    }

    const params = new URLSearchParams(options).toString();

    const http = request(`api/study/audit/list?${params}`, {
        signal, // 重点参数
        method: "GET",
        headers,
    });
    return { http, cancel };
};

// cache 缓存
const { http, cancel } = query(params);
this.cancelQuery = cancel; // 缓存 cancel 函数
http.then(()=>{
    this.cancelQuery = null // 清除缓存
}).catch(()=>{
    this.cancelQuery = null // 清除缓存
})

// callback 调用
if(this.cancelQuery instanceof Function)  this.cancelQuery();
```
