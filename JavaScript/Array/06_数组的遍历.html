    for (var i = 0, len = arr.length; i < len; i++) {       注意：这里的不存在是 属性不存在 
            if (  !arr[i]  )              continue; // 排除null 、undefined和不存在 的情况   代码的严格性 最好是写上这句话
            if (  arr[i] === undefined  ) continue; // 排除undefined和不存在 的情况
            if (  !(i in arr)  )          continue; // 排除不存在 的情况
        }
